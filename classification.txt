#Upload Dataset to Colab

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler
from sklearn.metrics import accuracy_score, confusion_matrix, classification_report

# Choose a direct built-in classifier (you can change this)
from sklearn.linear_model import LogisticRegression     # Option 1
# from sklearn.tree import DecisionTreeClassifier      # Option 2
# from sklearn.neighbors import KNeighborsClassifier   # Option 3
# from sklearn.naive_bayes import GaussianNB           # Option 4

data = pd.read_csv("your_file_name.csv")   # change filename
print(data.head())
print("Shape:", data.shape)

data = data.dropna()   # simple method

#By default, take the last column as target:
X = data.iloc[:, :-1]
y = data.iloc[:, -1]

#If teacher gives a named label column (example: "Class"):
# X = data.drop("Class", axis=1)
# y = data["Class"]

scaler = StandardScaler()
X = scaler.fit_transform(X)

X_train, X_test, y_train, y_test = train_test_split(X, y,
                                                    test_size=0.3,
                                                    random_state=42)

model = LogisticRegression()
model.fit(X_train, y_train)

y_pred = model.predict(X_test)

#PERFORMANCE EVALUATION

accuracy = accuracy_score(y_test, y_pred)
print("Accuracy:", accuracy)

cm = confusion_matrix(y_test, y_pred)
print("Confusion Matrix:\n", cm)

plt.figure(figsize=(6,4))
sns.heatmap(cm, annot=True, fmt="d", cmap="Blues")
plt.xlabel("Predicted")
plt.ylabel("Actual")
plt.title("Confusion Matrix")
plt.show()

print("Classification Report:\n")
print(classification_report(y_test, y_pred))

#Compare Multiple Algorithms
from sklearn.tree import DecisionTreeClassifier
from sklearn.neighbors import KNeighborsClassifier
from sklearn.naive_bayes import GaussianNB

models = {
    "Logistic": LogisticRegression(),
    "Decision Tree": DecisionTreeClassifier(),
    "KNN": KNeighborsClassifier(),
    "Naive Bayes": GaussianNB()
}

for name, m in models.items():
    m.fit(X_train, y_train)
    pred = m.predict(X_test)
    print(name, "Accuracy:", accuracy_score(y_test, pred))
